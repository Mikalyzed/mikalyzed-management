```typescript
// scripts/dms/decommission-readiness.ts
// Run ≥30 days after 0.D cutover. Exits 0 only if ALL decommission criteria met.
//   npx tsx scripts/dms/decommission-readiness.ts

import { prisma } from '@/lib/db'

interface Check { name: string; ok: boolean; detail: string }

async function main() {
  const checks: Check[] = []

  // CHECK 1: every InventoryVehicle row still has a VehicleMigrationMap entry
  const ivCount = await prisma.inventoryVehicle.count()
  const mappedIvCount = await prisma.vehicleMigrationMap.count({ where: { oldInventoryVehicleId: { not: null } } })
  checks.push({
    name: 'iv_to_map_complete',
    ok: ivCount === mappedIvCount,
    detail: `InventoryVehicle.count=${ivCount}, mapped=${mappedIvCount}`,
  })

  // CHECK 2: every canonical Vehicle that originated from IV has legacyInventoryVehicleId set
  const canonicalLinked = await prisma.vehicle.count({ where: { legacyInventoryVehicleId: { not: null } } })
  checks.push({
    name: 'canonical_audit_trail_complete',
    ok: canonicalLinked >= ivCount * 0.95, // allow some drift if backfill saw orphans become merged_into
    detail: `${canonicalLinked} canonical Vehicles carry legacyInventoryVehicleId vs ${ivCount} IVs`,
  })

  // CHECK 3: no Opportunity points at a missing Vehicle
  const orphanOpps = await prisma.opportunity.count({
    where: { vehicleId: { not: null }, vehicle: null },
  })
  checks.push({
    name: 'opportunity_attribution_intact',
    ok: orphanOpps === 0,
    detail: `${orphanOpps} Opportunities with broken vehicleId`,
  })

  // CHECK 4: cutover ActivityLog entry exists
  const cutover = await prisma.activityLog.findFirst({
    where: { action: 'reader_cutover_flipped' },
    orderBy: { createdAt: 'desc' },
  })
  const daysSinceCutover = cutover ? Math.floor((Date.now() - new Date(cutover.createdAt).getTime()) / 86400000) : -1
  checks.push({
    name: 'audit_window_elapsed',
    ok: daysSinceCutover >= 30,
    detail: `daysSinceCutover=${daysSinceCutover} (need ≥30)`,
  })

  // CHECK 5: signoff document exists with "signed off" text
  const fs = await import('fs')
  let signedOff = false
  try {
    const text = fs.readFileSync('.planning/phases/00-vehicle-identity-unification/decommission-signoff.md', 'utf-8')
    signedOff = /signed off/i.test(text) && /\b202[0-9]/.test(text) // has the magic phrase and a year
  } catch {}
  checks.push({
    name: 'user_signoff_present',
    ok: signedOff,
    detail: signedOff ? 'signoff file contains "signed off" and a year' : 'signoff file missing or unfilled',
  })

  // CHECK 6: legacy redirect endpoint has been hit at least once (best-effort — checked via ActivityLog if instrumented, else manual via Vercel logs)
  // (Soft check — not blocking. Document in stdout.)
  const legacyRedirectActivity = await prisma.activityLog.count({ where: { action: 'legacy_redirect_hit' } })
  console.log(`[decommission-readiness] (info only) legacy_redirect_hit ActivityLog entries: ${legacyRedirectActivity}`)

  // Render report
  console.log('[decommission-readiness] checks:')
  for (const c of checks) console.log(`  [${c.ok ? '✓' : '✗'}] ${c.name}: ${c.detail}`)
  const failed = checks.filter(c => !c.ok)
  if (failed.length) {
    console.error(`[decommission-readiness] BLOCKED: ${failed.length} check(s) failed`)
    process.exit(1)
  }
  console.log('[decommission-readiness] ✓ ALL CHECKS PASS — 0.E may proceed')
}

main().catch((e) => { console.error(e); process.exit(1) })
```
