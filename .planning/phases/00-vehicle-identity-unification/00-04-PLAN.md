```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/db'

export async function GET(req: NextRequest, { params }: { params: Promise<{ oldId: string }> }) {
  const { oldId } = await params
  const map = await prisma.vehicleMigrationMap.findFirst({
    where: { OR: [{ oldVehicleId: oldId }, { oldInventoryVehicleId: oldId }] },
    select: { canonicalVehicleId: true },
  })
  if (!map) return NextResponse.json({ error: 'Not found' }, { status: 404 })
  return NextResponse.redirect(new URL(`/vehicles/${map.canonicalVehicleId}`, req.url), 301)
}
```

**E. Create `_components/ActivityTab.tsx`**

Per CONTEXT: unified timeline from `/api/vehicles/[id]/history` (already reads ActivityLog for vehicle + parts). Filter chips for narrowing by event type. Newest first.

```tsx
'use client'
import { useState, useEffect } from 'react'
interface Event { id: string; action: string; entityType: string; details: any; createdAt: string; actorId: string | null }
const FILTER_CHIPS: { label: string; actions: string[] }[] = [
  { label: 'All', actions: [] },
  { label: 'Status', actions: ['status_changed', 'inventory_status_changed', 'recon_started', 'sold'] },
  { label: 'Parts', actions: ['part_added', 'part_received', 'part_installed'] },
  { label: 'Messages', actions: ['message_sent', 'message_received'] },
  { label: 'Migration', actions: ['canonical_backfill_merged', 'canonical_backfill_orphan_iv', 'canonical_backfill_orphan_v', 'canonical_manual_merge'] },
]
export default function ActivityTab({ vehicleId }: { vehicleId: string }) {
  const [active, setActive] = useState(0)
  const [events, setEvents] = useState<Event[]>([])
  useEffect(() => {
    const filter = FILTER_CHIPS[active].actions
    const qs = filter.length ? `?actions=${filter.join(',')}` : ''
    fetch(`/api/vehicles/${vehicleId}/history${qs}`).then(r => r.json()).then(setEvents)
  }, [vehicleId, active])
  return (
    <div>
      <div className="flex gap-2 mb-4 overflow-x-auto">
        {FILTER_CHIPS.map((c, i) => (
          <button key={c.label} onClick={() => setActive(i)} className={`px-3 py-1 rounded-full text-sm whitespace-nowrap ${i === active ? 'bg-primary text-primary-foreground' : 'bg-muted'}`}>{c.label}</button>
        ))}
      </div>
      <ul className="space-y-2">
        {events.map(e => (
          <li key={e.id} className="rounded-lg border p-3 text-sm">
            <div className="font-medium">{e.action.replace(/_/g, ' ')}</div>
            <div className="text-xs text-muted-foreground">{new Date(e.createdAt).toLocaleString()}</div>
            {e.details && <pre className="mt-1 text-xs text-muted-foreground">{JSON.stringify(e.details, null, 2)}</pre>}
          </li>
        ))}
        {events.length === 0 && <li className="text-sm text-muted-foreground">No activity yet</li>}
      </ul>
    </div>
  )
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/db'
import { requireRole } from '@/lib/auth'  // existing auth helper

export async function GET(req: NextRequest) {
  await requireRole(['admin'])  // admin-only per CONTEXT
  const url = new URL(req.url)
  const q = url.searchParams.get('q')?.trim().toLowerCase() ?? ''
  // Orphan IVs: canonical Vehicle created from IV with no recon (status='inventory_only')
  const orphanIVs = await prisma.vehicle.findMany({
    where: {
      status: 'inventory_only',
      priorVehicleId: null,
      ...(q ? { OR: [{ make: { contains: q, mode: 'insensitive' } }, { model: { contains: q, mode: 'insensitive' } }, { vin: { contains: q, mode: 'insensitive' } }] } : {}),
    },
    take: 100,
    orderBy: { createdAt: 'desc' },
  })
  // Orphan Vs: canonical Vehicle with no legacy IV link and not inventory_only
  const orphanVs = await prisma.vehicle.findMany({
    where: {
      legacyInventoryVehicleId: null,
      status: { not: 'inventory_only' },
      ...(q ? { OR: [{ make: { contains: q, mode: 'insensitive' } }, { model: { contains: q, mode: 'insensitive' } }, { vin: { contains: q, mode: 'insensitive' } }] } : {}),
    },
    take: 100,
    orderBy: { createdAt: 'desc' },
  })
  return NextResponse.json({ orphanIVs, orphanVs })
}

export async function POST(req: NextRequest) {
  const user = await requireRole(['admin'])
  const { orphanIvVehicleId, orphanVVehicleId, notes } = await req.json()
  // Strategy: merge IV scalars onto V; mark IV-side canonical Vehicle as status='merged_into' or delete (planner choice — recommend status flag for audit trail)
  const ivVehicle = await prisma.vehicle.findUnique({ where: { id: orphanIvVehicleId } })
  const vVehicle = await prisma.vehicle.findUnique({ where: { id: orphanVVehicleId } })
  if (!ivVehicle || !vVehicle) return NextResponse.json({ error: 'Not found' }, { status: 404 })

  await prisma.$transaction([
    prisma.vehicle.update({
      where: { id: vVehicle.id },
      data: {
        vehicleInfo: vVehicle.vehicleInfo ?? ivVehicle.vehicleInfo,
        mileage: vVehicle.mileage ?? ivVehicle.mileage,
        location: vVehicle.location ?? ivVehicle.location,
        askingPrice: vVehicle.askingPrice ?? ivVehicle.askingPrice,
        vehicleCost: vVehicle.vehicleCost ?? ivVehicle.vehicleCost,
        purchaseType: vVehicle.purchaseType ?? ivVehicle.purchaseType,
        purchasedFrom: vVehicle.purchasedFrom ?? ivVehicle.purchasedFrom,
        titleStatus: vVehicle.titleStatus ?? ivVehicle.titleStatus,
        dateInStock: vVehicle.dateInStock ?? ivVehicle.dateInStock,
        inventoryStatus: vVehicle.inventoryStatus ?? ivVehicle.inventoryStatus,
        legacyInventoryVehicleId: ivVehicle.legacyInventoryVehicleId,
      },
    }),
    prisma.vehicle.update({
      where: { id: ivVehicle.id },
      data: { status: 'merged_into' /* sentinel — never shown in normal lists */ },
    }),
    prisma.vehicleMigrationMap.create({
      data: {
        oldVehicleId: vVehicle.id,
        oldInventoryVehicleId: ivVehicle.legacyInventoryVehicleId,
        canonicalVehicleId: vVehicle.id,
        matchMethod: 'manual_review',
        matchConfidence: 'manual',
        notes: notes ?? null,
      },
    }),
    prisma.activityLog.create({
      data: {
        entityType: 'vehicle',
        entityId: vVehicle.id,
        action: 'canonical_manual_merge',
        actorId: user.id,
        details: { mergedFromVehicleId: ivVehicle.id, notes },
      },
    }),
  ])
  return NextResponse.json({ ok: true })
}
```

```tsx
'use client'
import { useState, useEffect } from 'react'
export default function VehicleMergePage() {
  const [q, setQ] = useState('')
  const [data, setData] = useState<{ orphanIVs: any[]; orphanVs: any[] }>({ orphanIVs: [], orphanVs: [] })
  const [selIv, setSelIv] = useState<string | null>(null)
  const [selV, setSelV] = useState<string | null>(null)
  const [notes, setNotes] = useState('')

  async function load() {
    const r = await fetch(`/api/admin/vehicle-merge?q=${encodeURIComponent(q)}`)
    setData(await r.json())
  }
  useEffect(() => { load() }, [q])

  async function merge() {
    if (!selIv || !selV) return
    await fetch('/api/admin/vehicle-merge', {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ orphanIvVehicleId: selIv, orphanVVehicleId: selV, notes }),
    })
    setSelIv(null); setSelV(null); setNotes(''); load()
  }

  return (
    <div className="...">
      <h1>Unmatched Vehicles</h1>
      <input value={q} onChange={e => setQ(e.target.value)} placeholder="Search YMM or VIN" className="..." />
      <div className="grid grid-cols-2 gap-4">
        <div>
          <h2>Orphan IVs (inventory-only)</h2>
          {data.orphanIVs.map(v => (
            <button key={v.id} onClick={() => setSelIv(v.id)} className={selIv === v.id ? 'bg-primary text-primary-foreground' : ''}>
              {v.year} {v.make} {v.model} — {v.stockNumber} — {v.vin}
            </button>
          ))}
        </div>
        <div>
          <h2>Orphan Vs (recon-only)</h2>
          {data.orphanVs.map(v => (
            <button key={v.id} onClick={() => setSelV(v.id)} className={selV === v.id ? 'bg-primary text-primary-foreground' : ''}>
              {v.year} {v.make} {v.model} — {v.stockNumber} — {v.vin}
            </button>
          ))}
        </div>
      </div>
      <textarea value={notes} onChange={e => setNotes(e.target.value)} placeholder="Notes (optional)" />
      <button disabled={!selIv || !selV} onClick={merge}>Merge selected</button>
    </div>
  )
}
```
