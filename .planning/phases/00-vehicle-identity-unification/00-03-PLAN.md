```typescript
function normalizePurchaseTypeForDualWrite(raw: string | null | undefined): string | null {
  if (!raw) return null
  const upper = raw.trim().toUpperCase().replace(/-/g, '_')
  if (upper === 'FLOORING') return 'PURCHASED'
  if (upper === 'PURCHASED') return 'PURCHASED'
  if (upper === 'CONSIGNMENT') return 'CONSIGNMENT'
  if (upper === 'TRADE_IN' || upper === 'TRADEIN') return 'TRADE_IN'
  return upper
}
```

**Part B: Create `scripts/dms/dual-write-diff-report.ts`**

```typescript
// scripts/dms/dual-write-diff-report.ts
// Run daily during the 7-day dual-write observation window:
//   npx tsx scripts/dms/dual-write-diff-report.ts
//
// Flags any drift between InventoryVehicle and canonical Vehicle aliases.
// Outputs a structured report; if non-zero drift, EXIT 1 so a cron alert fires.

import { prisma } from '@/lib/db'

interface Drift {
  stockNumber: string | null
  field: string
  ivValue: unknown
  canonicalValue: unknown
  ivId: string
  canonicalVehicleId: string
}

async function main() {
  const drift: Drift[] = []

  const ivs = await prisma.inventoryVehicle.findMany({
    select: {
      id: true, stockNumber: true, vehicleInfo: true, mileage: true, location: true,
      askingPrice: true, vehicleCost: true, purchaseType: true, purchasedFrom: true,
      titleStatus: true, dateInStock: true, status: true, vin: true,
    },
  })

  for (const iv of ivs) {
    if (!iv.stockNumber) continue
    const canonical = await prisma.vehicle.findUnique({
      where: { stockNumber: iv.stockNumber },
      select: {
        id: true, vehicleInfo: true, mileage: true, location: true,
        askingPrice: true, vehicleCost: true, purchaseType: true, purchasedFrom: true,
        titleStatus: true, dateInStock: true, inventoryStatus: true, vin: true,
      },
    })

    if (!canonical) {
      drift.push({
        stockNumber: iv.stockNumber, field: 'canonical_missing',
        ivValue: iv.id, canonicalValue: null, ivId: iv.id, canonicalVehicleId: '',
      })
      continue
    }

    // Compare key inventory scalars
    const checks: [string, unknown, unknown][] = [
      ['vehicleCost', iv.vehicleCost, canonical.vehicleCost],
      ['askingPrice', iv.askingPrice, canonical.askingPrice],
      ['mileage', iv.mileage, canonical.mileage],
      ['location', iv.location, canonical.location],
      ['titleStatus', iv.titleStatus, canonical.titleStatus],
      ['dateInStock', iv.dateInStock?.toISOString() ?? null, canonical.dateInStock?.toISOString() ?? null],
      ['status_vs_inventoryStatus', iv.status, canonical.inventoryStatus],
    ]
    for (const [field, a, b] of checks) {
      if (a !== b) {
        drift.push({
          stockNumber: iv.stockNumber, field, ivValue: a, canonicalValue: b,
          ivId: iv.id, canonicalVehicleId: canonical.id,
        })
      }
    }
  }

  const report = {
    ranAt: new Date().toISOString(),
    inventoryVehicleCount: ivs.length,
    driftCount: drift.length,
    drift: drift.slice(0, 100), // truncate for readability
  }
  console.log(JSON.stringify(report, null, 2))
  if (drift.length > 0) {
    console.error(`[dual-write-diff] ${drift.length} drift(s) detected`)
    process.exit(1)
  }
}

main().catch((e) => { console.error(e); process.exit(1) })
```

User runs this daily during the 7-day dual-write window. Zero drift for 5+ consecutive days = green to proceed to 0.D.
